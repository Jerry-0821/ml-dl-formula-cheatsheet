\section{Backpropagation}

\subsection{Backward Pipeline}

\FormulaName{Dense-layer reverse flow}
\[
dA^{[l]} \to dZ^{[l]} \to dW^{[l]}, db^{[l]} \to dA^{[l-1]}.
\]
\Use{Reverse-mode differentiation for dense layers.}

\subsection{Output Layer Simplifications}

\FormulaName{Sigmoid output with binary cross-entropy}
\[
dZ^{[L]} = A^{[L]} - Y.
\]
\Shapes{$A^{[L]}, Y, dZ^{[L]} \in \R^{1 \times m}$.}
\Use{Binary output-layer gradient.}

\FormulaName{Softmax output with categorical cross-entropy}
\[
dZ^{[L]} = A^{[L]} - Y.
\]
\Shapes{$A^{[L]}, Y, dZ^{[L]} \in \R^{K \times m}$.}
\Use{Multi-class output-layer gradient.}
\Notes{Same compact form, different output dimension.}

\subsection{General Hidden-Layer Backpropagation}

\FormulaName{Activation gradient}
\[
dZ^{[l]}
=
dA^{[l]} \hadamard g'^{[l]}(Z^{[l]}).
\]
\Shapes{$dA^{[l]}, dZ^{[l]}, Z^{[l]} \in \R^{n^{[l]} \times m}$.}
\Use{Hidden-layer pre-activation gradient.}

\FormulaName{Previous-activation gradient}
\[
dA^{[l-1]}
=
(W^{[l]})\transpose dZ^{[l]}.
\]
\[
W^{[l]} \in \R^{n^{[l]} \times n^{[l-1]}},
\qquad
dA^{[l-1]} \in \R^{n^{[l-1]} \times m}.
\]
\Use{Gradient passed to previous layer.}

\FormulaName{Shape check for $dA^{[l-1]}$}
\[
(W^{[l]})\transpose dZ^{[l]}
:
(n^{[l-1]} \times n^{[l]})
\cdot
(n^{[l]} \times m)
\to
n^{[l-1]} \times m.
\]
\Use{Backpropagation dimension check.}

\subsection{Parameter Gradients}

\FormulaName{Weight gradient}
\[
dW^{[l]}
=
\frac{1}{m}
dZ^{[l]}(A^{[l-1]})\transpose.
\]
\Shapes{$dW^{[l]} \in \R^{n^{[l]} \times n^{[l-1]}}$.}
\Use{Dense-layer weight gradient.}

\FormulaName{Bias gradient}
\[
db^{[l]}
=
\frac{1}{m}
\sum_{i=1}^{m}
dZ^{[l](i)}
\in
\R^{n^{[l]} \times 1}.
\]
\Use{Dense-layer bias gradient.}
\Notes{$dZ^{[l](i)}$ is the $i$-th column of $dZ^{[l]}$.}

\FormulaName{Shape check for $dW^{[l]}$}
\[
dZ^{[l]}(A^{[l-1]})\transpose
:
(n^{[l]} \times m)
\cdot
(m \times n^{[l-1]})
\to
n^{[l]} \times n^{[l-1]}.
\]
\Use{Parameter-gradient dimension check.}

\subsection{Shape Check and Parameter Update Link}

\FormulaName{Basic gradient descent update}
\[
\begin{aligned}
W^{[l]} &\leftarrow W^{[l]} - \alpha dW^{[l]},\\
b^{[l]} &\leftarrow b^{[l]} - \alpha db^{[l]}.
\end{aligned}
\]
\Shapes{$W^{[l]},dW^{[l]} \in \R^{n^{[l]} \times n^{[l-1]}}$; $b^{[l]},db^{[l]} \in \R^{n^{[l]} \times 1}$.}
\Use{Basic gradient descent update.}

\subsection{Gradient Checking}

\FormulaName{Numerical gradient approximation}
\[
\frac{\partial \cost}{\partial \theta_i}
\approx
\frac{
\cost(\theta+\epsilon e_i)
-
\cost(\theta-\epsilon e_i)
}{
2\epsilon
}.
\]
\Shapes{$\theta_i$ is one scalar parameter; $e_i$ is the matching basis vector.}
\Use{Sanity check for backprop implementation.}

\FormulaName{Vector-form approximation}
\[
g_{\mathrm{approx}}
\approx
\frac{
\cost(\theta+\epsilon)
-
\cost(\theta-\epsilon)
}{
2\epsilon
}.
\]
\Shapes{$g_{\mathrm{approx}}$ has the same shape as $g$.}
\Use{Numerical gradient vector.}

\FormulaName{Relative difference}
\[
\operatorname{diff}
=
\frac{
\left\lVert g-g_{\mathrm{approx}}\right\rVert_2
}{
\left\lVert g\right\rVert_2
+
\left\lVert g_{\mathrm{approx}}\right\rVert_2
}.
\]
\Use{Gradient-check discrepancy.}
\Notes{$\epsilon$ small; disable dropout / randomness.}
